## 🎯 目标

1. 保留单页分类和长图分类两种实现，通过**配置文件**选择使用哪种分类方式

4. 为后续的切片功能和完整单元形成做数据准备

```python
class SheetType(Enum):

    RIGHT_PAGE = "right"      # 右页（A3 右半部分）
    FULL_PAGE = "full"        # 整页（A4 或 A3 整张）

    """

    Attributes:
        set_id: 套编号，唯一标识一套试卷（如 "SET_20260320_001"）
        sheet_id: 张编号，物理层面的试卷纸编号（如 "SHEET_001"）

    sheet_id: str

    original_image: str = ""
    corrected_image: str = ""
    bbox_in_stitched: Optional[list] = None
    
    def to_dict(self) -> dict:
        return {
            'set_id': self.set_id,
            'sheet_id': self.sheet_id,
            'order': self.order,
            'original_order': self.original_order,
            'sheet_type': self.sheet_type.value if self.sheet_type else None,
            'layout': self.layout.value if self.layout else None,
            'original_image': self.original_image,
            'corrected_image': self.corrected_image,
```

```python
import uuid
from datetime import datetime

def generate_set_id() -> str:
    """
    生成唯一的套编号
    
    格式：SET_YYYYMMDD_XXXX
    - SET: 固定前缀
    - YYYYMMDD: 日期
    - XXXX: UUID 的后 4 位
    
    示例：SET_20260320_A1B2
    """
    date_str = datetime.now().strftime("%Y%m%d")
    unique_suffix = uuid.uuid4().hex[:4].upper()
    return f"SET_{date_str}_{unique_suffix}"
```

```python
from PIL import Image

def detect_sheet_layout(image_path: str) -> str:
    """
    检测试卷纸的布局类型
    
    Args:
        image_path: 图片路径
        
    Returns:
        LayoutType 枚举值
        
    判断逻辑：
    1. 检测图片宽高比
    2. 如果 width/height ≈ 2 (A3 横向)，返回 LEFT_RIGHT
    3. 如果 width/height ≈ 0.5 (A3 纵向)，需要进一步判断左右页
    4. 否则返回 FULL_PAGE (A4)
    """
    with Image.open(image_path) as img:
        width, height = img.size
        aspect_ratio = width / height
        
        if 1.8 <= aspect_ratio <= 2.2:
            # A3 横向展开
            return LayoutType.LEFT_RIGHT.value
        elif aspect_ratio < 0.7:
            # A3 纵向，可能是左页或右页
            # TODO: 通过内容分析判断是左页还是右页
            return LayoutType.LEFT_PAGE.value  # 暂时默认左页
        else:
            # A4 或接近正方形
            return LayoutType.FULL_PAGE.value
```

```python
def run_long_image_classification(workspace_dir: str, image_path_manager, db_session) -> dict:
    """
    长图分类（使用套 - 张编号系统）
    """
    # 1. 从步骤 1 获取矫正结果和元数据
    correction_output = load_correction_output(workspace_dir)
    set_id = correction_output['set_id']
    sheet_metadata_list = correction_output['sheet_metadata']
    
    # 2. 按任意顺序拼接成长图（拼接顺序不影响最终顺序判断）
    stitched_image_path = stitch_images(
        sheet_metadata_list, 
        output_path=os.path.join(workspace_dir, 'stitched_images', f'{set_id}_stitched.jpg')
    )
    
    # 3. 调用大模型 API
    prompt = """你是一个经验丰富的判卷老师。这是一套试卷的图片，包含 N 张试卷纸（已拼接成一张长图），请你分析每张试卷纸类型。
    
    可选三种类型：
    1. 题目页
    2. 答题纸
    3. 题目和答题混合页
    
    ## 重要概念
    
    **一张试卷纸**：
    - 指的是物理上的一张纸（A3 或 A4）
    - 在长图中，每张试卷纸占据一个独立的矩形区域
    - 你需要识别每张试卷纸的**左右边界**
    
    ## 重要判断规则
    
    **核心逻辑**：
    - 如果这套试卷有**专门的答题纸**，那么：    
      - 其他页面上即使有"题号 + 手写答案"且专门答题纸中有上述页面中的同样题号，上述页面也应该判断为**题目页**
      - 因为一张试卷中的题目标号在答题纸中存在，说明学生的手写内容只是**草稿/标注**，正式答案应该写在答题纸上        
    
    **判断优先级**：
    1. 先找答题纸
    2. 如果找到答题纸，其他页面即使有手写答案，但题号在答题纸中存在，也优先判断为题目页
    3. 只有当**没有答题纸**时，"题目 + 手写答案"才判断为混合页
    
    ## 张顺序判断
    
    **重要**：请根据试卷内容的逻辑顺序（如页码、题号连续性等）判断每张试卷纸的正确顺序，而不是根据图片的排列顺序。
    
    用 JSON 格式输出每页的类型、顺序号和判断理由。"""
    
    api_response = call_api(stitched_image_path, prompt)
    
    # 4. 解析返回结果
    classification_results = parse_api_response(api_response)
    
    # 5. 关联元数据（使用大模型返回的顺序号，而不是输入顺序）
    for result in classification_results:
        # 找到对应的试卷纸元数据（通过 bbox 匹配）
        sheet_meta = find_sheet_by_bbox(sheet_metadata_list, result['bbox'])
        if sheet_meta:
            sheet_meta['sheet_type'] = result['type']
            sheet_meta['order'] = result['order']  # 使用大模型返回的顺序号
            sheet_meta['reason'] = result['reason']
            sheet_meta['bbox_in_stitched'] = result['bbox']
    
    # 6. 按照大模型判断的顺序号重新排序
    sheet_metadata_list.sort(key=lambda x: x['order'])
    
    # 7. 保存结果
    output = {
        'set_id': set_id,
        'stitched_image_path': stitched_image_path,
        'sheets': sheet_metadata_list
    }
    
    save_classification_output(workspace_dir, output)
    
    return output
```

```python
    """
    基于套 - 张编号系统，创建完整单元
    
    判断逻辑：
    1. 根据 classification_output 中的 sheet_type，确定哪些张是题目页，哪些是答题纸
    2. 根据 layout_output 中的题目坐标和答案坐标，创建切片
    3. 通过题号关联，将题目切片和答案切片组合成完整单元
    4. 每个完整单元都记录 set_id 和 sheet_associations，便于追溯
    """
    complete_units = []
    
    # 示例逻辑
    for sheet in classification_output['sheets']:
        if sheet['sheet_type'] == 'question_page':
            # 从题目页提取题目切片
            question_slices = extract_question_slices(sheet)
            
            for q_slice in question_slices:
                # 查找答题纸中对应题号的答案
                answer_slices = find_answer_slices(
                    q_slice.question_numbers,
                    classification_output['sheets']
                )
                
                # 创建完整单元
                unit = CompleteUnit(
                    unit_id=f"{sheet['set_id']}_UNIT_{q_slice.question_numbers[0]}",
                    set_id=sheet['set_id'],
                    question_number=q_slice.question_numbers[0],
                    question_slice=q_slice,
                    answer_slices=answer_slices,
                    sheet_associations=[sheet['sheet_id']] + [a.sheet_id for a in answer_slices]
                )
                
                complete_units.append(unit)
    
    return complete_units
```
